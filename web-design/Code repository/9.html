# js模拟new实现

new 运算符用来创建自定义对象类型或具有构造函数的内置对象类型的实例

new运算符做了这些事:

  1. 创建一个空对象
  2. 将这个对象链接到原型链上
  3. 绑定this的指向
  4. 返回这个对象
function objectFactory() {
    var obj = new Object(),
        constructor = [].shift.call(arguments);
    Object.setPrototypeOf(obj, Constructor.prototype);
    var ret = Constructor.apply(obj, arguments);
    return typeof ret === 'object' ? ret : obj;
}

# call, apply, bind模拟实现

call

Function.prototype.call = function(context) {
    var context = context || window;
    context.fn = this;

    var args = [];
    for (var i = 1; len = arguments.length; i < len; i++) {
        args.push('arguments[' + i + ']');
    }

    var result = eval('context.fn(' + args + ')');

    delete context.fn
    return result;
}

apply

Function.apply = function(context, arr) {
    var context = Object(context) || window;
    context.fn = this;

    var result;
    if (!arr) {
        result = context.fn();
    } else {
        var args = [];
        for (var i = 0, len = arr.length; i < len; i++) {
            args.push('arr[' + i + ']');
        }
        result = eval('context.fn(' + args + ')')
    }
    delete context.fn
    return result;
}

bind

Function.prototype.bind = function(context) {
    if (typeof this !== "function") {
        throw new Error("Function.prototype.bind -what is trying to be bound is not callable");
    }
    var self = this;
    var args = Array.prototype.slice.call(arguments, 1);

    var fNOP = function() {};

    var fBound = function() {
        var bindArgs = Array.prototype.slice.call(arguments);
        return self.apply(this instanceof fNop ? this : context, args.concat(bindArgs));
    }
    fNop.prototype = this.prototype;
    fBound.prototype = new fNOP();
    return fBound;
}

  1. 模拟栈结构
function flattenDeep(arr) {
    const result = []
    // 将数组元素拷贝至栈，直接赋值会改变原数组
    const stack = [...arr]
    // 如果栈不为空，则循环遍历
    while (stack.length !== 0) {
        const val = stack.pop()
        if (Array.isArray(val)) {
            // 如果是数组再次入栈，并且展开了一层
            stack.push(...val)
        } else {
            // 如果不是数组，就用头插法插入到结果数组中
            result.unshift(val)
        }
    }
    return result
}

# 数组去重

  1. set
  2. reduce
function unique(arr) {
    return arr.sort().reduce((acc, cur) => {
        if (acc.length === 0 || acc[acc.length - 1] !== cur) {
            acc.push(cur);
        }
        return acc
    }, [])
}
  1. filter
function unique(arr) {
    return arr.filter((element, index, array) => {
        return array.indexOf(element) === index
    })
}
  1. 对象属性唯一(此方法存在问题,无法区别字符串和数字,true跟"true"也会视为相同)
function unique(arr) {
    var obj = new Object();
    var array = [];
    arr.forEach((item, index) => {
        if (!obj[item]) {
            obj[item] = 1;
            array.push(item);
        }
        obj[item] += 1;
    });
    return array;
}

# 实现一个Promise

function Promise(fn) {
    this.cbs = [];

    const resolve = (value) => {
        setTimeout(() => {
            this.data = value;
            this.cbs.forEach((cb) => cb(value));
        });
    }

    fn(resolve);
}

Promise.prototype.then = function(onResolved) {
    return new Promise((resolve) => {
        this.cbs.push(() => {
            const res = onResolved(this.data);
            if (res instanceof Promise) {
                res.then(resolve);
            } else {
                resolve(res);
            }
        });
    });
};

一个promise应接受一个函数作为参数, 这个函数接受 resolve，reject 作为参数，这里不考虑错误情况仅接受一个 resolve ，通过 resolve 内部实现异步，将 value 值直接挂载到当前 promise 实例上并依次执行回调函数。

接下来需要实现链式调用， promise 原型上定义 then 方法，只需要返回一个 promise 的实例即可再次调用， then 通常接受两个函数作为参数，第一个参数用来处理 fulfilled 状态，第二个处理 rejected 状态，此处同样只考虑一种情况，这里 onResolved 就是成功返回后的处理函数。这个处理函数会接受一个参数即前一个 promise resolve 的结果。

接下来关键在于then内部实现

 return new Promise((resolve) => {
        this.cbs.push(() => {//此处this指的是调用then方法的Promise对象，因为箭头函数的原因
            const res = onResolved(this.data);
            if (res instanceof Promise) {
                res.then(resolve);//由用户决定何时返回,并将返回的值通过resolve挂载到then新生成的promise对象上便于下一个then方法使用
            } else {
                resolve(res);
            }
        });
    })

返回一个新的promise对象，同时这个promise执行传入的函数时会向 第一个promise对象this.cbs这个数组中放入一个cb ，result用来保存传入then方法中的函数onResolved执行后的结果。(onResolve接受的参数就是我们正常使用then时.then((data)=>{console.log(data)})拿到的结果，这里已经被我们挂载到promise.data上了)

最后把结果挂载到新生成的promise实例上用于之后then的继续调用,如果onResolved返回的是用户编写的promise对象则通过then(resolve)出去,普通值则直接resolve。
(resolve是构造函数Promise生成,所以每个promise实例在生成resolve传入fn中后resolve的this都指向自身的promise)(关键在于箭头函数的this指向的是 定义箭头函数的上下文而不是普通函数执行的上下文对象)

最后看一个例子加深理解

new Promise((resolve) => {
  setTimeout(() => {
    // resolve1
    resolve(1);
  }, 500);
})
  // then1
  .then((res) => {
    console.log(res);
    // user promise
    return new Promise((resolve) => {
      setTimeout(() => {
        // resolve2
        resolve(2);
      }, 500);
    });
  })
  // then2
  .then(console.log);
